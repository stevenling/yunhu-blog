# 冒泡排序

# 一、思路

每相邻两个数都比较，最大的就沉在最右边，内循环的范围一次一次减少。

代码，因为一个一个沉下去的就是排好的，相邻元素顺序错误我们就交换。

# 二、过程

# 三、代码

#include <stdio.h>
int main()
{
    // n 为需要排序的数的数量
    int a[100], i, n, j, temp; 

    scanf("%d", &n);

    for (i = 0; i < n; i++)
    {
        scanf("%d", &a[i]);
    }
        
    for (i = 0; i < n; i++)
    {
        // 一次外循环结束沉淀一个数比较的那些数就减少
        for (j = 0; j < n - 1 - i; j++) 
        {
            // 后面一个数小于前面的则交换
            if (a[j] > a[j + 1]) 
            {
                temp = a[j];
                a[j] = a[j + 1];
                a[j + 1] = temp;
            }
        }
    }

    // 输出排序后的结果
    for (i = 0; i < n; i++) 
    {
        printf("%d ", a[i]);
    }      
}
